	\section{ADT Map}
	empty: -> map\\
	add: map, key -> val \\
	del: map, key -> map \\
	add(empty, k, v) = m \\
	k==k' => add(add(m,k',v'),k,v) = add(m,k,v) \\
	k!=k' => add(add(m,k',v'),k,v) = add(add(m,k,v),k',v') \\
	\\
	get(add(m,k,v),k) = v \\
	k!=k' => get(add(m,k',v), k) = get(m,k) \\
	\\
	del(add(m,k,v), k) = m \\
	k!=k' => del(add(m,k',v), k) =  del(m,k) \\
